```javascript
const output = $json.output || '';
try {
  // Check if output is JSON tool call
  const stripped = output.replace(/^```(?:json)?\n?/m, '').replace(/\n?```$/m, '').trim();
  if (stripped.startsWith('{') && stripped.includes('"tool"')) {
    const decision = JSON.parse(stripped);
    return [{json: {
      ...$('Resolve Model').item.json,  // preserve context
      tool_name: decision.tool,
      tool_input: decision.input || {},
      acknowledge_message: decision.acknowledge || null,
      direct_response: null
    }}];
  }
} catch(e) {}
// Direct response — no tool
return [{json: {

  tool_name: null,
  tool_input: null,
  acknowledge_message: null,

}}];
```
