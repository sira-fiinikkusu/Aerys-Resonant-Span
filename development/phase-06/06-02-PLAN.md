```javascript
// STRATEGY: replace — rebuilt from inputs on every message
const fs = require('fs');
const soulMd = fs.readFileSync('/home/node/aerys-config/soul.md', 'utf8');
const intermediateSteps = JSON.parse($json.intermediateSteps || '[]');

// Format tool return data for polisher awareness
let toolContext = '';
if (intermediateSteps.length > 0) {
  toolContext = '## Tool Results\n' + intermediateSteps.map(step => {
    const obs = typeof step.observation === 'string'
      ? step.observation.substring(0, 2000)
      : JSON.stringify(step.observation).substring(0, 2000);
    return `[${step.action?.tool}]: ${obs}`;
  }).join('\n\n');
}

const systemPrompt = [
  `## Background\n${soulMd}`,
  `## Core Agent Response\n${$json.output}`,
  toolContext,
  $json.memory_context ? `## Relevant Memories\n${$json.memory_context}` : '',
  $json.thread_context ? `## Recent Conversation\n${$json.thread_context}` : '',
  $json.profile_context ? `## Person Profile\n${$json.profile_context}` : '',
  `## Rules`,
  `- Rewrite the Core Agent Response in Aerys's voice. Preserve all factual content.`,
  `- If the response references tool results, incorporate them naturally.`,
  $json.conversation_privacy === 'public'
    ? `- PUBLIC CHANNEL: Scrub specific PII from your output (phone numbers, addresses, SSNs, medical details). Generalize sensitive facts. Names and general preferences are fine.`
    : `- PRIVATE DM: No PII restrictions. Share all information freely.`,
  `- Do NOT add preambles like "Here's my rewrite." Just output the polished response directly.`,
  `- Keep the response length similar to the Core Agent's. Do not pad.`,
].filter(Boolean).join('\n\n');

return [{json: {
  ...$json,
  polisher_system_prompt: systemPrompt,
  intermediateSteps // parsed back for downstream use
}}];
```
