**New node: "Format Trace" (Code node) — on the true branch:**

```javascript
// Crabwalk-style trace — NO user content, NO Aerys response text (privacy)
// STRATEGY: replace — rebuilt from intermediateSteps on every message
const steps = $json.intermediateSteps || [];
const model = ($json.model_tier || 'unknown').charAt(0).toUpperCase() + ($json.model_tier || 'unknown').slice(1);
const startTs = $json._start_ts || Date.now();
const elapsed = ((Date.now() - startTs) / 1000).toFixed(1);

const TOOL_ICONS = {
  media: '[IMG]',
  research: '[RES]',
  email: '[EML]',
};

let lines = [`[${model}] ${elapsed}s`];

if (steps.length === 0) {
  lines.push('+-- Direct response');
} else {
  let totalTokens = 0;
  steps.forEach((step, i) => {
    const isLast = i === steps.length - 1;
    const prefix = isLast ? '+--' : '|--';
    const icon = TOOL_ICONS[step.action?.tool] || '[?]';
    const tokens = step.tokenUsage?.totalTokens || 0;
    totalTokens += tokens;
    const tokenStr = tokens > 0 ? ` ${tokens} tokens` : '';
    lines.push(`${prefix} ${icon} ${step.action?.tool || 'unknown'}${tokenStr}`);
  });
  if (totalTokens > 0) {
    lines.push(`+-- Total: ${totalTokens.toLocaleString()} tokens`);
  }
}

return [{json: {
  trace_message: '```\n' + lines.join('\n') + '\n```',
  debug_channel_id: 'YOUR_DEBUG_CHANNEL_ID' // #aerys-debug — hardcoded, no $vars
}}];
```
