```javascript
// Snowflake to Date helper
function snowflakeToDate(id) {
  try { return new Date(Number((BigInt(id) >> 22n)) + 1420070400000); }
  catch { return null; }
}

const items = $input.all();
const msgs = items.map(i => i.json);

// Get the triggering message context from Normalize Message node
const trigMsg = $('Normalize Message').item.json;
const trigId = trigMsg.message_id || trigMsg.id;
const triggerTime = trigMsg.timestamp ? new Date(trigMsg.timestamp) : new Date();

function nameOf(m) {
  return m.author?.global_name || m.author?.username || m.username || 'Unknown';
}
function msgDate(m) {
  if (m.timestamp) return new Date(m.timestamp);
  if (m.id) return snowflakeToDate(m.id);
  return null;
}

// Filter: all messages (humans AND bots), before trigger time, not the triggering message.
// Bots are intentionally included — Aerys should see her own prior responses in the room
// and can hear other bots (e.g. other bots) as [BotName]: message entries. The nameOf() function
// uses author.global_name/username so bot names surface naturally. If noisy utility bots
// (level bots, music bots) become a problem, add a blacklist of bot user IDs here.
const roomMsgs = msgs.filter(m => {
  const hasText = (m.content ?? '').trim().length > 0;
  const mt = msgDate(m);
  const beforeTrigger = mt ? (mt <= triggerTime) : true;
  const notWake = m.id !== trigId;
  return hasText && beforeTrigger && notWake;
});

// Participant summary header
const speakerCounts = {};
roomMsgs.forEach(m => {
  const name = nameOf(m);
  speakerCounts[name] = (speakerCounts[name] || 0) + 1;
});
const participantSummary = Object.keys(speakerCounts).length > 0
  ? `PARTICIPANTS: ${Object.entries(speakerCounts).sort((a,b)=>b[1]-a[1]).map(([n,c])=>`${n} (${c})`).join(', ')}\n---\n`
  : '';

// Speaker-tagged transcript (capped at 5800 chars)
const lines = roomMsgs.map(m => `[${nameOf(m)}]: ${m.content.trim()}`);
const transcript = (participantSummary + lines.join('\n')).slice(0, 5800);

return [{ json: { thread_context: transcript || '' } }];
```

4. **Format Member List** (Code node) — receives output from Get Server Members:
```javascript
const items = $input.all();
const members = items.map(i => i.json);
const memberMap = {};
members.forEach(m => {
  const userId = m.user?.id || m.id;
  const displayName = m.nick || m.user?.global_name || m.user?.username || 'Unknown';
  if (userId) memberMap[userId] = displayName;
});
const memberListText = Object.entries(memberMap)
  .map(([id, name]) => `<@${id}>: ${name}`)
  .join('\n');
return [{ json: { member_list: memberListText, member_map: memberMap } }];
```
