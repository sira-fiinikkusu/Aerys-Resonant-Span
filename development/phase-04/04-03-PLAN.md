const inputData = $('Webhook Trigger').first().json;

if (rows.length === 0) {
  // Cold start — no profile yet
  return [{
    json: {
      profile: {
        display_name: null,
        lines: [],
        cold_start: true
      }
    }
  }];

// Group claims by category for readable injection
const categoryOrder = ['basic', 'interests', 'relationship', 'emotional', 'other'];
const grouped = {};
rows.forEach(r => {
  const cat = r.key_label.split('.')[0] || 'other';
  if (!grouped[cat]) grouped[cat] = [];
  grouped[cat].push(`• ${r.claim_text}`);
});

const lines = [];
categoryOrder.forEach(cat => {
  if (grouped[cat]) lines.push(...grouped[cat]);
});
// Add any remaining categories not in the order
Object.keys(grouped).filter(k => !categoryOrder.includes(k)).forEach(cat => {
  lines.push(...grouped[cat]);
});

// Find display name if claimed
const nameClaim = rows.find(r => r.key_label === 'basic.name' || r.key_label === 'name');
const displayName = nameClaim ? nameClaim.claim_text.split(': ').pop() : null;

return [{
  json: {
    profile: {
      display_name: displayName,
      lines,
      cold_start: false
    }
  }
}];

4. **Fetch Promotion Candidates** (Postgres node, executeQuery):
```sql
WITH aggregated AS (
  SELECT
    speaker_id,
    key_label,
    MAX(model_conf)         AS max_model_conf,
    COUNT(*)                AS total_repeat_count,
    MIN(first_seen)         AS first_seen_ts,
    MAX(last_seen)          AS last_seen_ts,
    EXTRACT(EPOCH FROM (MAX(last_seen) - MIN(first_seen))) / 86400 AS span_days,
    MAX(CASE WHEN asserted_by = 'self' THEN 1 ELSE 0 END) AS has_self_assertion,
    STRING_AGG(value_text, ' ||| ' ORDER BY last_seen DESC) AS all_values,
    STRING_AGG(id::text, ',' ORDER BY last_seen DESC) AS row_ids
  FROM userinfo
  GROUP BY speaker_id, key_label
),
scored AS (
  SELECT *,
    LEAST(1.0,
      0.70 * COALESCE(max_model_conf, 0.5)
      + 0.20 * LEAST(1.0, LN(1 + total_repeat_count) / LN(4))
      + CASE WHEN has_self_assertion = 1 THEN 0.20 ELSE 0.05 END
      + CASE WHEN total_repeat_count > 2 THEN 0.10 ELSE 0.0 END
    ) AS computed_confidence
  FROM aggregated
)
SELECT * FROM scored
WHERE total_repeat_count >= 2
  AND (
    (has_self_assertion = 1 AND computed_confidence >= 0.75)
    OR
    (has_self_assertion = 0 AND computed_confidence >= 0.85 AND span_days >= 7)
  );
```
